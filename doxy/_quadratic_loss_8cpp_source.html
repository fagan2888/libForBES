57 int QuadraticLoss::call(Matrix& x, double& f) {
58  f = 0.0;
59  for (size_t j = 0; j < x.getNrows(); j++) {
60  double fi;
61  fi = x[j];
62  if (!m_is_zero_p) {
63  fi -= m_p->getData()[j];
64  }
65  fi *= fi;
66  if (!m_is_uniform_weights) {
67  fi *= m_w->getData()[j];
68  }
69  f += fi;
70  }
71  if (m_is_uniform_weights) {
72  f *= m_uniform_w;
73  }
74  f /= 2.0;
75  return ForBESUtils::STATUS_OK;
76 }

78 int QuadraticLoss::call(Matrix& x, double& f, Matrix& grad) {
79  f = 0.0;
80  for (size_t j = 0; j < x.getNrows(); j++) {
81  double fi;
82  double gi;
83  fi = x[j];
84  if (!m_is_zero_p) {
85  fi -= m_p->getData()[j];
86  }
87  gi = fi;
88  fi *= fi;
89  if (!m_is_uniform_weights) {
90  double w = m_w->getData()[j];
91  fi *= w;
92  gi *= w;
93  }
94  f += fi;
95  grad[j] = gi;
96  }
97  if (m_is_uniform_weights) {
98  f *= m_uniform_w;
99  }
100  f /= 2.0;
101  return ForBESUtils::STATUS_OK;
102 }

104 int QuadraticLoss::callConj(Matrix& x, double& f_star) {
105  f_star = 0.0;
106  for (size_t i = 0; i < x.getNrows(); i++) {
107  f_star += x[i]*(2.0 * (m_is_zero_p ? 0.0 : m_p->get(i))
108  + (x[i] / (m_is_uniform_weights ? m_uniform_w : m_w->get(i))));
109  }
110  f_star /= 2.0;
111  return ForBESUtils::STATUS_OK;
112 }

114 int QuadraticLoss::callConj(Matrix& x, double& f_star, Matrix& grad) {
115  f_star = 0.0;
116  for (size_t i = 0; i < x.getNrows(); i++) {
117  double gradi;
118  double pi;
119  pi = m_is_zero_p ? 0.0 : m_p->get(i);
120  gradi = pi + x[i] / (m_is_uniform_weights ? m_uniform_w : m_w->get(i));
121  grad.set(i, 0, gradi);
122  f_star += x[i]*(gradi + pi);
123  }
124  f_star /= 2.0;
125  return ForBESUtils::STATUS_OK;
126 }
