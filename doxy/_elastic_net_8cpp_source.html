30 int ElasticNet::call(Matrix& x, double& f) {
31  //LCOV_EXCL_START
32  if (!x.isColumnVector()) {
33  throw std::invalid_argument("x must be a column-vector");
34  }
35  //LCOV_EXCL_STOP
36  f = 0.0;
37  for (size_t i = 0; i < x.getNrows(); i++) {
38  double xi;
39  xi = x[i];
40  f += m_mu * std::abs(xi) + (m_lambda / 2.0) * std::pow(xi, 2);
41  }
42  return ForBESUtils::STATUS_OK;
43 }

45 int ElasticNet::callProx(Matrix& x, double gamma, Matrix& prox, double& g_at_prox) {
46  //LCOV_EXCL_START
47  if (!x.isColumnVector()) {
48  throw std::invalid_argument("x must be a column-vector");
49  }
50  //LCOV_EXCL_STOP
51  double gm = gamma * m_mu;
52  double alpha = 1 + m_lambda * gamma; // alpha > 0 [assuming gamma>0 and lambda>0].
53  g_at_prox = 0.0;
54  for (size_t i = 0; i < x.getNrows(); i++) {
55  double xi;
56  double yi;
57  xi = x[i];
58  yi = max(0.0, abs(xi) - gm) / alpha;
59  prox[i] = (xi < 0 ? -1 : 1) * yi;
60  g_at_prox += m_mu * yi + (m_lambda / 2.0) * std::pow(yi, 2);
61  }
62  return ForBESUtils::STATUS_OK;
63 }

65 int ElasticNet::callProx(Matrix& x, double gamma, Matrix& prox) {
