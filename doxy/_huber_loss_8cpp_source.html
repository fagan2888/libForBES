31 int HuberLoss::call(Matrix& x, double& f) {
32  //LCOV_EXCL_START
33  if (!x.isColumnVector()) {
34  throw std::invalid_argument("x must be a column-vector");
35  }
36  //LCOV_EXCL_STOP
37  f = 0.0;
38  for (size_t i = 0; i < x.getNrows(); i++) {
39  double xi;
40  xi = x[i];
41  if (std::abs(xi) <= m_delta) {
42  f += std::pow(xi, 2) / (2.0 * m_delta);
43  } else {
44  f += std::abs(xi) - m_delta / 2.0;
45  }
46  }
47  return ForBESUtils::STATUS_OK;
48 }

50 int HuberLoss::call(Matrix& x, double& f, Matrix& grad) {
51  //LCOV_EXCL_START
52  if (!x.isColumnVector()) {
53  throw std::invalid_argument("x must be a column-vector");
54  }
55  //LCOV_EXCL_STOP
56  f = 0.0;
57  for (size_t i = 0; i < x.getNrows(); i++) {
58  double xi;
59  xi = x[i];
60  if (std::abs(xi) <= m_delta) {
61  f += std::pow(xi, 2) / (2.0 * m_delta);
62  grad[i] = xi / m_delta;
63  } else {
64  f += std::abs(xi) - m_delta / 2.0;
65  grad[i] = std::signbit(xi) ? -1.0 : 1.0;
66  }
67  }
68  return ForBESUtils::STATUS_OK;
69 }
