46 int ConjugateFunction::callProx(Matrix& x, double gamma, Matrix& prox) {
47  double one_over_gamma = 1.0 / gamma;
48  Matrix x_over_gamma = x;
49  x_over_gamma *= one_over_gamma;
50  int status = m_function.callProx(x_over_gamma, one_over_gamma, prox);
51  prox *= gamma;
52  for (size_t i = 0; i < x.getNrows(); i++) {
53  prox.set(i, 0, x.get(i, 0) - prox.get(i, 0));
54  }
55  return status;
56 }

58 int ConjugateFunction::callProx(Matrix& x, double gamma, Matrix& prox, double& f_at_prox) {
59  int status = callProx(x, gamma, prox);
60  if (ForBESUtils::STATUS_OK != status) {
61  return status;
62  }
63  return call(prox, f_at_prox);
