74 int Quadratic::call(Matrix& x, double& f) {
75  if (!m_is_Q_eye) {
76  if (m_is_q_zero) {
77  f = m_Q->quad(x);
78  } else {
79  f = m_Q->quad(x, *m_q);
80  }
81  } else {
82  f = (x * x).get(0, 0);
83  if (!m_is_q_zero) {
84  f += ((*m_q) * x).get(0, 0);
85  }
86  }
87  return ForBESUtils::STATUS_OK;
88 }

90 int Quadratic::call(Matrix& x, double& f, Matrix& grad) {
91  int statusComputeGrad = computeGradient(x, grad); // compute the gradient of f at x (grad)
92  if (statusComputeGrad != ForBESUtils::STATUS_OK) {
93  return statusComputeGrad;
94  }
95  // f = (1/2)*(grad+q)'*x
96  f = ((m_is_q_zero ? grad : grad + (*m_q)) * x).get(0, 0) / 2;
97  return ForBESUtils::STATUS_OK;

112 int Quadratic::callConj(Matrix& y, double& f_star, Matrix& g) {
113  Matrix z = (m_is_q_zero || m_q == NULL) ? y : y - *m_q; // z = y
114  if (m_is_Q_eye || m_Q == NULL) {
115  g = z;
116  f_star = (z * z).get(0, 0);
117  return ForBESUtils::STATUS_OK;
118  }
119  if (m_Q != NULL && Matrix::MATRIX_DIAGONAL == m_Q->getType()) {
120  /* Q is diagonal */
121  g = z;
122  f_star = 0.0;
123  for (size_t i = 0; i < z.getNrows(); i++) {
124  g[i] /= m_Q->get(i, i);
125  f_star += z[i] * g[i];
126  }
127  return ForBESUtils::STATUS_OK;
128  }
129 
130  if (m_solver == NULL) {
131  m_solver = new CholeskyFactorization(*m_Q);
132  int status = m_solver->factorize();
133  if (0 != status) {
134  return ForBESUtils::STATUS_NUMERICAL_PROBLEMS;
135  }
136  }
137 
138  m_solver->solve(z, g); // Q*g = z OR g = Q \ z
139  f_star = (z * g).get(0, 0); // fstar = z' *g
140  return ForBESUtils::STATUS_OK;
141 }

159 int Quadratic::callProx(Matrix& v, double gamma, Matrix& prox) {
160  // (I+gamma Q)^{-1}(v-gamma q)
161  int status;
162  CGSolver * solver = NULL;
163  if (!m_is_Q_eye) {
164  // If Q is not I, we need to create a CGSolver for (I + gamma Q)
165  Matrix Q_tilde(*m_Q);
166  size_t n = m_Q->getNrows();
167  static Matrix Eye = MatrixFactory::MakeIdentity(n, 1.0);
168  status = Matrix::add(Q_tilde, 1.0, Eye, gamma);
169  if (!ForBESUtils::is_status_ok(status)) {
170  return status;
171  }
172  MatrixOperator Q_tilde_op(Q_tilde);
173  Matrix P(n, n, Matrix::MATRIX_DIAGONAL);
174  for (size_t i = 0; i < n; i++) {
175  P[i] = 1 / Q_tilde.get(i, i);
176  }
177  MatrixOperator P_op(P);
178  solver = new CGSolver(Q_tilde_op, P_op);
179  }
180  /*
181  * v_gamma_b = v - gamma * b
182  */
183  Matrix v_gamma_b = MatrixFactory::ShallowVector();
184  if (m_is_q_zero) {
185  v_gamma_b = MatrixFactory::ShallowMatrix(v);
186  } else {
187  v_gamma_b = v;
188  status = Matrix::add(v_gamma_b, -gamma, *m_q, 1.0);
189  if (!ForBESUtils::is_status_ok(status)) {
190  return status;
191  }
192  }
193  if (!m_is_Q_eye) {
194  status = solver->solve(v_gamma_b, prox);
195  if (!ForBESUtils::is_status_ok(status)) {
196  return status;
197  }
198  } else {
199  prox = v_gamma_b;
200  }
201  if (solver != NULL) {
202  delete solver;
203  }
204  return ForBESUtils::STATUS_OK;
205 
206 }
