148 bool FunctionOntologicalClass::is_quadratic() {
149  string quad_name = FunctionOntologyRegistry::quadratic().getName();
150  if (getName().compare(quad_name) == 0) return true;
151  bool gauge = false;
152  std::list<FunctionOntologicalClass> superclasses = getSuperclasses();
153  for (std::list<FunctionOntologicalClass>::iterator cl = superclasses.begin();
154  cl != superclasses.end();
155  ++cl) {
156  if ((*cl).getName().compare(quad_name) == 0) {
157  gauge = true;
158  break;
159  }
160  }
161  return gauge;
162 }
