51 int CGSolver::solve(Matrix& b, Matrix& solution) {
52  m_iterations_count = 0;
53  m_err = NAN;
54  Matrix r(b);
55  Matrix z = m_precond->call(r);
56  Matrix p = MatrixFactory::ShallowVector(z, 0);
57  bool keepgoing = true;
58  Matrix Ap;
59  while (keepgoing) {
60  Ap = m_linop->call(p); // Ap = A * p
61  double a_denom = (p * Ap).get(0, 0); //
62  double a_numer = (r * z).get(0, 0); //
63  double alpha = a_numer / a_denom; // alpha = (r,z)/(p, Ap);
64  Matrix::add(solution, alpha, p, 1.0); // x = x + alpha * p
65  Matrix r_new = r; // r_new = r
66 
67  m_err = LAPACKE_dlange(
68  LAPACK_COL_MAJOR,
69  'I',
70  r_new.getNrows(),
71  1,
72  r_new.getData(),
73  r_new.getNrows()); // compute ||r||_{inf}
74 
75  if (m_err < m_tolerance) {
76  break; // stop if tolerance reached
77  }
78 
79  Matrix::add(r_new, -alpha, Ap, 1.0); // r_new = r - alpha A p;
80  Matrix z_new = m_precond->call(r_new); // z_new = P(r_new)
81  double zr = (r_new * z_new).get(0, 0);
82  double beta = zr / a_numer; // beta = (r_new, z_nwq)/(r, z)
83  Matrix::add(p, 1.0, z_new, beta); // p = beta p + z_new
84  z = z_new; // z = z_new
85  r = r_new; // r = r_new
86  m_iterations_count++; // k = k + 1
87  if (m_iterations_count > m_max_iterations) {
88  keepgoing = false;
89  }
90  }
91  if (m_iterations_count>=m_max_iterations){
92  return ForBESUtils::STATUS_MAX_ITERATIONS_REACHED;
93  }
94  return ForBESUtils::STATUS_OK;
95 }
