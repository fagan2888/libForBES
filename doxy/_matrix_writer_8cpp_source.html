39 void MatrixWriter::printJSON(FILE* fp) {
40  fprintf(fp, "{\n");
41  fprintf(fp, " \"%s\":\"%s\",\n", MATRIX_TYPE, m_matrix.getTypeString().c_str());
42  fprintf(fp, " \"%s\":"FMT_SIZE_T",\n", MATRIX_NROWS, m_matrix.getNrows());
43  fprintf(fp, " \"%s\":"FMT_SIZE_T",\n", MATRIX_NCOLS, m_matrix.getNcols());
44  if (m_matrix.getType() != Matrix::MATRIX_SPARSE) {
45  fprintf(fp, " \"%s\":"FMT_SIZE_T",\n", MATRIX_DATALENGTH, m_matrix.length());
46  } else {
47  fprintf(fp, " \"%s\":"FMT_SIZE_T",\n", MATRIX_NZ, m_matrix.m_triplet->nnz);
48  }
49 
50  fprintf(fp, " \"%s\":%d,\n", MATRIX_ENFORCE_DENSE_MODE, m_enforceDenseMode);
51 
52  fprintf(fp, " \"%s\":[", MATRIX_DATA);
53  if (m_matrix.getType() == Matrix::MATRIX_DENSE || m_enforceDenseMode) {
54  for (size_t j = 0; j < m_matrix.getNcols(); j++) {
55  for (size_t i = 0; i < m_matrix.getNrows(); i++) {
56  fprintf(fp, "%g", m_matrix.get(i, j));
57  if (!(i == m_matrix.getNrows() - 1 && j == m_matrix.getNcols() - 1)) {
58  fprintf(fp, ", ");
59  }
60  }
61  }
62  } else if (m_matrix.getType() != Matrix::MATRIX_SPARSE) {
63  for (size_t i = 0; i < m_matrix.length(); i++) {
64  fprintf(fp, "%g", m_matrix.m_data[i]);
65  if (i != m_matrix.length() - 1) {
66  fprintf(fp, ", ");
67  }
68  }
69  } else if (m_matrix.getType() == Matrix::MATRIX_SPARSE) {
70  for (size_t i = 0; i < m_matrix.m_triplet->nnz; i++) {
71  fprintf(fp, "[%d, %d, %g]",
72  (static_cast<int*> (m_matrix.m_triplet->i))[i],
73  (static_cast<int*> (m_matrix.m_triplet->j))[i],
74  (static_cast<double*> (m_matrix.m_triplet->x))[i]);
75  if (i != m_matrix.m_triplet->nnz - 1) {
76  fprintf(fp, ", ");
77  }
78  }
79  }
80  fprintf(fp, "]\n}");
81 }
