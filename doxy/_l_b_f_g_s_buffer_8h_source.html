203  inline long get_k_minus_j(size_t j) {
204  register long t = m_idx - j;
205  if (t >= 0) {
206  return t;
207  } else if (m_current_mem < m_mem || j >= m_mem) {
208  return LBFGS_BUFFER_EOB;
209  } else {
210  return (m_mem + t);
211  }
212  }
