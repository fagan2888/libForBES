37 int HingeLoss::call(Matrix& x, double& f) {
38  if (!x.isColumnVector()) {
39  throw std::invalid_argument("x must be a column-vector");
40  }
41  f = 0.0;
42  for (size_t i = 0; i < x.getNrows(); i++) {
43  double si = 1 - m_b->get(i) * x[i];
44  if (si > 0) {
45  f += si;
46  }
47  }
48  f *= m_mu;
49  return ForBESUtils::STATUS_OK;
50 }

52 int HingeLoss::callProx(Matrix& x, double gamma, Matrix& prox) {
53  if (!x.isColumnVector()) {
54  throw std::invalid_argument("x must be a column-vector");
55  }
56  double gm = gamma*m_mu;
57  for (size_t i = 0; i < x.getNrows(); i++) {
58  double bi;
59  double bxi;
60  bi = m_b->get(i);
61  bxi = bi * x[i];
62  if (bxi < 1) {
63  prox[i] = bi * std::min(1.0, bxi + gm);
64  } else {
65  prox[i] = x[i];
66  }
67  }
68  return ForBESUtils::STATUS_OK;
69 }

71 int HingeLoss::callProx(Matrix& x, double gamma, Matrix& prox, double& f_at_prox) {
72  if (!x.isColumnVector()) {
73  throw std::invalid_argument("x must be a column-vector");
74  }
75  double gm = gamma*m_mu;
76  f_at_prox = 0.0;
77  for (size_t i = 0; i < x.getNrows(); i++) {
78  double si;
79  double pi;
80  double bi;
81  double bxi;
82  bi = m_b->get(i);
83  bxi = bi * x[i];
84  if (bxi < 1) {
85  pi = bi * std::min(1.0, bxi + gm);
86  } else {
87  pi = x[i];
88  }
89  si = 1 - m_b->get(i) * pi;
90  if (si > 0) {
91  f_at_prox += si;
92  }
93  prox[i] = pi;
94  }
95  f_at_prox *= m_mu;
96  return ForBESUtils::STATUS_OK;
97 }
