44 int SumOfNorm2::call(Matrix& x, double& f) {
45  f = 0.0;
46  const size_t n = x.getNrows();
47  if (n % m_partition_length != 0) {
48  throw std::invalid_argument("Given vector cannot be partitioned");
49  }
50  /* number of chunks */
51  const size_t n_chunks = n / m_partition_length;
52  int status = ForBESUtils::STATUS_OK;
53  for (size_t j = 0; j < n_chunks && status == ForBESUtils::STATUS_OK; j++) {
54  double f_temp;
55  Matrix x_part = MatrixFactory::ShallowVector(x, m_partition_length, j * m_partition_length);
56  status = m_norm2->call(x_part, f_temp);
57  f += f_temp;
58  }
59  f *= m_mu;
60  return status;
61 }

63 int SumOfNorm2::callProx(Matrix& v, double gamma, Matrix& prox) {
64  const size_t n = v.getNrows();
65  if (n % m_partition_length != 0) {
66  throw std::invalid_argument("Given vector cannot be partitioned");
67  }
68 
69  const size_t n_chunks = n / m_partition_length;
70  Matrix v_part = MatrixFactory::ShallowVector();
71 
72  int status = ForBESUtils::STATUS_OK;
73  for (size_t j = 0; j < n_chunks && status == ForBESUtils::STATUS_OK; j++) {
74  double norm_temp;
75  v_part = MatrixFactory::ShallowVector(v, m_partition_length, j * m_partition_length);
76  Matrix prox_part = MatrixFactory::ShallowVector(prox, m_partition_length, j * m_partition_length);
77  status = m_norm2->call(v_part, norm_temp);
78  double factor = 1 - (gamma * m_mu / norm_temp);
79  for (size_t i = 0; i < m_partition_length; i++) {
80  prox_part[i] = factor * v_part[i];
81  }
82  }
83  return status;
84 }

86 int SumOfNorm2::callProx(Matrix& v, double gamma, Matrix& prox, double& f_at_prox) {
87  const size_t n = v.getNrows();
88  if (n % m_partition_length != 0) {
89  throw std::invalid_argument("Given vector cannot be partitioned");
90  }
91 
92  f_at_prox = 0.0;
93  const size_t n_chunks = n / m_partition_length;
94  Matrix v_part = MatrixFactory::ShallowVector();
95 
96  int status = ForBESUtils::STATUS_OK;
97  for (size_t j = 0; j < n_chunks && status == ForBESUtils::STATUS_OK; j++) {
98  double norm_temp;
99  v_part = MatrixFactory::ShallowVector(v, m_partition_length, j * m_partition_length);
100  Matrix prox_part = MatrixFactory::ShallowVector(prox, m_partition_length, j * m_partition_length);
101  status = m_norm2->call(v_part, norm_temp);
102  double factor = std::max<double>(0.0, 1.0 - (gamma * m_mu / norm_temp));
103  for (size_t i = 0; i < m_partition_length; i++) {
104  prox_part[i] = factor * v_part[i];
105  }
106  f_at_prox += std::abs(factor * norm_temp);
107  }
108  f_at_prox *= m_mu;
109  return status;
110 }

121 int SumOfNorm2::dualNorm(Matrix& x, double& norm) {
122  /* this implementation is very much similar to #call */
123  norm = 0.0;
124  const size_t n = x.getNrows();
125  if (n % m_partition_length != 0) {
126  throw std::invalid_argument("Given vector cannot be partitioned");
127  }
128  /* number of chunks */
129  const size_t n_chunks = n / m_partition_length;
130  for (size_t j = 0; j < n_chunks; j++) {
131  double f_temp;
132  Matrix x_part = MatrixFactory::ShallowVector(x, m_partition_length, j * m_partition_length);
133  m_norm2->call(x_part, f_temp);
134  norm = std::max(norm, f_temp);
135  }
136  norm /= m_mu;
137  return ForBESUtils::STATUS_OK;
138 }
