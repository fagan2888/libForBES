46 int OpLTI::call(Matrix& y, double alpha, Matrix& u, double gamma) {
47  // y := gamma * y + alpha * T(u)
48  // T(u) = [x1, x2, ..., xN]
49  if (!u.isColumnVector()) {
50  throw std::invalid_argument("u should be a column vector");
51  }
52  if (u.getNrows() != (m_N - 1) * m_B.getNcols()) {
53  std::ostringstream oss;
54  oss << "u: wrong dimension - should be (N-1)*n_u = "
55  << (m_N - 1) << "*" << m_B.getNcols()
56  << "=" << ((m_N - 1) * m_B.getNcols());
57  throw std::invalid_argument(oss.str().c_str());
58  }
59  size_t n = m_A.getNrows();
60  size_t m = m_B.getNcols();
61  const size_t zero = 0;
62  Matrix y0 = MatrixFactory::ShallowVector(y, n, zero);
63  Matrix u0 = MatrixFactory::ShallowVector(u, m, zero);
64  int status = Matrix::mult(y0, alpha, m_B, u0, gamma);
65  ForBESUtils::fail_on_error(status);
66  std::cout << std::endl;
67  for (size_t i = 1; i < m_N - 1; i++) {
68  Matrix ui = MatrixFactory::ShallowVector(u, m, i * m);
69  std::cout << "u[" << i << "] = " << ui << std::endl;
70  }
71  return ForBESUtils::STATUS_UNDEFINED_FUNCTION;
72 }
