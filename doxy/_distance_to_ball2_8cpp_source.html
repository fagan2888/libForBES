37 int DistanceToBall2::call(Matrix& x, double& f, Matrix& grad) {
38  double norm_x_minus_c;
39  f = 0.0;
40  grad = x;
41  if (m_center != NULL) {
42  grad -= *m_center;
43  }
44  norm_x_minus_c = std::sqrt((grad * grad)[0]);
45  if (norm_x_minus_c > m_rho) {
46  f = 0.5 * m_w * std::pow(norm_x_minus_c - m_rho, 2);
47  grad *= (1-m_rho/norm_x_minus_c);
48  } else {
49  for (size_t i = 0; i < grad.getNrows(); i++){
50  grad[i] = 0.0;
51  }
52  }
53  if (std::abs(m_w - 1.0) > 1e-14) {
54  grad *= m_w;
55  }
56  return ForBESUtils::STATUS_OK;
57 }
