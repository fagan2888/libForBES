46 int CholeskyFactorization::factorize() {
47  if (m_matrix_type == Matrix::MATRIX_SPARSE) {
48  /* Cholesky decomposition of a SPARSE matrix: */
49  if (m_matrix->m_sparse == NULL) {
50  m_matrix->_createSparse();
51  }
52  /* analyze */
53  m_factor = cholmod_analyze(m_matrix->m_sparse, Matrix::cholmod_handle());
54  /* factorize */
55  cholmod_factorize(m_matrix->m_sparse, m_factor, Matrix::cholmod_handle());
56  /* Success: status = 0, else 1*/
57  return (m_factor->minor == m_matrix->m_nrows) ? ForBESUtils::STATUS_OK : ForBESUtils::STATUS_NUMERICAL_PROBLEMS;
58  } else { /* If this is any non-sparse matrix: */
59  memcpy(m_L, m_matrix->getData(), m_matrix->length() * sizeof (double)); /* m_L := m_matrix.m_data */
60  int info = ForBESUtils::STATUS_OK;
61  if (m_matrix_type == Matrix::MATRIX_DENSE) { /* This is a dense matrix */
62  info = LAPACKE_dpotrf(LAPACK_COL_MAJOR, 'L', m_matrix_nrows, m_L, m_matrix_nrows);
63 #ifdef SET_L_OFFDIAG_TO_ZERO
64  for (size_t i = 0; i < m_matrix_nrows; i++) {
65  for (size_t j = i + 1; j < m_matrix_nrows; j++) {
66  L.set(i, j, 0.0);
67  }
68  }
69 #endif
70  } else if (m_matrix_type == Matrix::MATRIX_SYMMETRIC) { /* This is a symmetric matrix */
71  info = LAPACKE_dpptrf(LAPACK_COL_MAJOR, 'L', m_matrix_nrows, m_L);
72  }
73  return info;
74  }
75 }

77 int CholeskyFactorization::solve(Matrix& rhs, Matrix& solution) {
78  if (m_matrix_type == Matrix::MATRIX_SPARSE) {
79  cholmod_dense *x;
80 
81  /* cast the RHS as a cholmod_dense b = rhs */
82  if (rhs.m_type == Matrix::MATRIX_DENSE) {
83 
84  cholmod_dense *b;
85  b = cholmod_allocate_dense(rhs.m_nrows, rhs.m_ncols, rhs.m_nrows, CHOLMOD_REAL, Matrix::cholmod_handle());
86  b->x = rhs.m_data;
87 
88  /* Solve - rhs is dense*/
89  x = cholmod_solve(CHOLMOD_A, m_factor, b, Matrix::cholmod_handle());
90  solution = Matrix(rhs.m_nrows, rhs.m_ncols);
91  solution.m_delete_data = false;
92  memcpy(solution.m_data, static_cast<double*> (x->x), rhs.m_nrows * rhs.m_ncols * sizeof (double));
93  cholmod_free_dense(&x, Matrix::cholmod_handle());
94 
95  } else if (rhs.m_type == Matrix::MATRIX_SPARSE) {
96  // still untested!
97  cholmod_sparse * rhs_sparse;
98  if (rhs.m_sparse == NULL) {
99  const_cast<Matrix&> (rhs)._createSparse();
100  }
101  rhs_sparse = rhs.m_sparse;
102  cholmod_sparse * result = cholmod_spsolve(CHOLMOD_LDLt, m_factor, rhs_sparse, Matrix::cholmod_handle());
103  solution = Matrix(rhs.m_nrows, rhs.m_ncols, Matrix::MATRIX_SPARSE);
104  solution.m_sparse = result;
105  solution._createTriplet();
106  } else {
107  throw std::logic_error("Not supported");
108  }
109  return ForBESUtils::STATUS_OK;
110  } else { /* the matrix to be factorized is not sparse */
111  int info = ForBESUtils::STATUS_UNDEFINED_FUNCTION;
112  solution = Matrix(rhs);
113  if (m_matrix_type == Matrix::MATRIX_DENSE) {
114  info = LAPACKE_dpotrs(LAPACK_COL_MAJOR, 'L', m_matrix_nrows, rhs.m_ncols, m_L, m_matrix_nrows, solution.m_data, m_matrix_nrows);
115  } else if (m_matrix_type == Matrix::MATRIX_SYMMETRIC) {
116  info = LAPACKE_dpptrs(LAPACK_COL_MAJOR, 'L', m_matrix_nrows, rhs.m_ncols, m_L, solution.m_data, m_matrix_nrows);
117  } else {
118  throw std::invalid_argument("This matrix type is not supported - only DENSE, SPARSE and SYMMETRIC are supported");
119  }
120  return info;
121  }
