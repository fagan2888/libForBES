44 Matrix MatrixFactory::MakeRandomSparse(size_t nrows, size_t ncols, size_t nnz, float offset, float scale) {
45  Matrix R = MakeSparse(nrows, ncols, nnz, Matrix::SPARSE_UNSYMMETRIC);
46  std::set<nice_pair> s;
47  nice_pair p;
48  while (true) { // construct pairs
49  p.first = (std::rand() % nrows);
50  p.second = (std::rand() % ncols);
51  s.insert(p);
52  if (s.size() == nnz) {
53  break;
54  }
55  }
56  for (std::set<nice_pair>::iterator it = s.begin(); it != s.end(); ++it) {
57  double rand;
58  rand = offset + (scale * std::rand()) / RAND_MAX;
59  R.set(it->first, it->second, rand);
60  }
61  return R;
62 }

64 Matrix MatrixFactory::MakeRandomMatrix(size_t nrows, size_t ncols, float offset, float scale, Matrix::MatrixType type) {
65  size_t len = 0;
66  switch (type) {
67  case Matrix::MATRIX_DENSE:
68  len = nrows * ncols;
69  break;
70  case Matrix::MATRIX_LOWERTR:
71  case Matrix::MATRIX_SYMMETRIC:
72  len = nrows * (nrows + 1) / 2;
73  break;
74  case Matrix::MATRIX_DIAGONAL:
75  len = nrows;
76  break;
77  case Matrix::MATRIX_SPARSE:
78  return MakeRandomSparse(nrows, ncols, std::ceil((nrows * ncols) / 3), offset, scale);
79  break;
80  default:
81  throw std::logic_error("unsupported");
82  }
83  Matrix mat(nrows, ncols, type);
84  for (size_t j = 0; j < len; j++) {
85  mat[j] = static_cast<double> (offset + (scale * std::rand()) / RAND_MAX);
86  }
87  return mat;
88 }

94 Matrix MatrixFactory::MakeSparse(size_t nrows, size_t ncols, size_t max_nnz, Matrix::SparseMatrixType stype) {
95  if (max_nnz > nrows * ncols) {
96  std::ostringstream oss;
97  oss << "Matrix " << nrows << "x" << ncols << "(max_size=" << (nrows*ncols)
98  << " cannot allocate " << max_nnz << "non-zeros";
99  throw std::invalid_argument(oss.str().c_str());
100  }
101  Matrix matrix(nrows, ncols, Matrix::MATRIX_SPARSE);
102  matrix.m_triplet = cholmod_allocate_triplet(nrows, ncols, max_nnz, stype, CHOLMOD_REAL, Matrix::cholmod_handle());
103  matrix.m_sparseStorageType = Matrix::CHOLMOD_TYPE_TRIPLET;
104  return matrix;
105 }

138 Matrix MatrixFactory::ShallowVector(const Matrix& orig, size_t size, size_t offset) {
139  if (orig.getType() != Matrix::MATRIX_DENSE) {
140  throw std::invalid_argument("This method can only be applied to dense vectors");
141  }
142  if (orig.getNcols() != 1 && orig.getNrows() != 1) {
143  throw std::invalid_argument("This method can only be applied to vectors");
144  }
145  Matrix v_shallow = Matrix(true);
146  v_shallow.m_transpose = orig.m_transpose;
147  v_shallow.m_nrows = orig.m_transpose ? 1 : size;
148  v_shallow.m_ncols = orig.m_transpose ? size : 1;
149  v_shallow.m_dataLength = v_shallow.m_nrows * v_shallow.m_ncols;
150  v_shallow.m_delete_data = false;
151  v_shallow.m_data = orig.m_data + offset;
152  return v_shallow;
153 }
