36 inline double vecNorm2(Matrix& x) {
37  double norm_x = 0.0;
38  for (size_t i = 0; i < x.getNrows(); i++) {
39  norm_x += std::pow(x[i], 2);
40  }
41  norm_x = std::sqrt(norm_x);

65 int Norm2::callProx(Matrix& x, double gamma, Matrix& prox) {
66  //LCOV_EXCL_START
67  if (!x.isColumnVector()) {
68  throw std::invalid_argument("x must be a column-vector");
69  }
70  //LCOV_EXCL_STOP
71  double norm_x = vecNorm2(x);
72  double gm = gamma*m_mu;
73  if (norm_x > gm) {
74  double s = 1 - gm / norm_x;
75  prox = s * x;
76  }
77  return ForBESUtils::STATUS_OK;

80 int Norm2::callProx(Matrix& x, double gamma, Matrix& prox, double& f_at_prox) {
81  //LCOV_EXCL_START
82  if (!x.isColumnVector()) {
83  throw std::invalid_argument("x must be a column-vector");
84  }
85  //LCOV_EXCL_STOP
86  double norm_x = vecNorm2(x);
87  double gm = gamma*m_mu;
88  if (norm_x > gm) {
89  double s = 1 - gm / norm_x;
90  for (size_t i = 0; i < x.getNrows(); i++) {
91  prox.set(i, 0, s * x[i]); // prox = s * x;
92  }
93  f_at_prox = m_mu * s * norm_x;
94  } else {
95  f_at_prox = 0.0;
96  }
97  return ForBESUtils::STATUS_OK;
98 }
