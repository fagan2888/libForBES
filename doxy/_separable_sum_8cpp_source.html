34 void prepare_cx(std::vector<size_t> * c_idx, const Matrix& x, Matrix * c_x) {
35  size_t k = 0;
36 
37  /* construct the sub-matrix */
38  for (std::vector<size_t>::iterator idx_iterator = c_idx->begin(); idx_iterator != c_idx->end(); ++idx_iterator) {
39  c_x -> set(k, 0, x.get(*idx_iterator, 0));
40  k++;
41  }
42 }

44 int SeparableSum::call(Matrix& x, double& f) {
45  //LCOV_EXCL_START
46  if (!x.isColumnVector()) {
47  throw std::invalid_argument("x must be a column-vector");
48  }
49  //LCOV_EXCL_STOP
50 
51  /* sub-matrix */
52  Matrix *c_x = NULL;
53 
54  /* initialization of result f(x) */
55  f = 0.0;
56 
57 
58  /* Invoke all functions */
59  for (std::map<Function*, std::vector<size_t> * >::iterator map_iterator = m_fun_idx_map.begin()
60  ; map_iterator != m_fun_idx_map.end()
61  ; ++map_iterator) {
62  /* current vector of indices*/
63  std::vector<size_t> * c_idx = NULL;
64 
65  /* temporary value of sub-invocations */
66  double f_temp;
67 
68  /* status of sub-invocations */
69  int status;
70 
71  /* Pointer to current function */
72  Function * c_fun = NULL;
73  c_fun = map_iterator->first;
74 
75  /* Function indices */
76  c_idx = map_iterator->second;
77 
78  /* initialize new sub-matrix */
79  c_x = new Matrix(c_idx->size(), 1);
80 
81  /* iterator for the indices of c_fun */
82  prepare_cx(c_idx, x, c_x);
83 
84  /* invoke sub-function on c_x, return f_temp */
85  status = c_fun -> call(*c_x, f_temp);
86  //LCOV_EXCL_START
87  if (ForBESUtils::STATUS_OK != status) {
88  delete c_x;
89  c_x = NULL;
90  return status;
91  }
92  //LCOV_EXCL_STOP
93  f += f_temp;
94  }
95 
96  if (c_x != NULL) {
97  delete c_x;
98  c_x = NULL;
99  }
100  return ForBESUtils::STATUS_OK;
101 }

107 int SeparableSum::callProx(Matrix& x, double gamma, Matrix& prox) {
108  //LCOV_EXCL_START
109  if (!x.isColumnVector()) {
110  throw std::invalid_argument("x must be a column-vector");
111  }
112  //LCOV_EXCL_STOP
113 
114  Matrix *c_x = NULL;
115  Matrix *c_prox = NULL;
116 
117  for (std::map<Function*, std::vector<size_t> * >::iterator map_iterator = m_fun_idx_map.begin()
118  ; map_iterator != m_fun_idx_map.end()
119  ; ++map_iterator) {
120  Function * c_fun = map_iterator->first;
121  std::vector<size_t> * c_idx = map_iterator->second;
122  c_x = new Matrix(c_idx->size(), 1);
123  prepare_cx(c_idx, x, c_x);
124 
125  c_prox = new Matrix(c_idx->size(), 1);
126  int status = c_fun -> callProx(*c_x, gamma, *c_prox);
127  //LCOV_EXCL_START
128  if (ForBESUtils::STATUS_OK != status) {
129  delete c_x;
130  delete c_prox;
131  return status;
132  }
133  //LCOV_EXCL_STOP
134  size_t k = 0;
135  std::vector<size_t>::iterator idx_iterator;
136  for (idx_iterator = c_idx->begin(); idx_iterator != c_idx->end(); ++idx_iterator) {
137  prox.set(*idx_iterator, 0, c_prox->get(k, 0));
138  k++;
139  }
140  }
141  delete c_x;
142  delete c_prox;
143  return ForBESUtils::STATUS_OK;
144 }

150 int SeparableSum::callConj(Matrix& x, double& f_star) {
151  //LCOV_EXCL_START
152  if (!x.isColumnVector()) {
153  throw std::invalid_argument("x must be a column-vector");
154  }
155  //LCOV_EXCL_STOP
156  f_star = 0.0;
157 
158  std::map<Function*, std::vector<size_t> * >::iterator map_iterator;
159  Function * c_fun = NULL;
160  Matrix *c_x = NULL;
161 
162  for (map_iterator = m_fun_idx_map.begin(); map_iterator != m_fun_idx_map.end(); ++map_iterator) {
163  std::vector<size_t> * c_idx = NULL;
164  c_fun = map_iterator->first; // current function
165  c_idx = map_iterator->second; // current index set I
166  c_x = new Matrix(c_idx->size(), 1); // prepare current vector
167  prepare_cx(c_idx, x, c_x);
168 
169  double f_star_temp;
170  int status = c_fun -> callConj(*c_x, f_star_temp);
171  //LCOV_EXCL_START
172  if (ForBESUtils::STATUS_OK != status) {
173  delete c_x;
174  return status;
175  }
176  f_star += f_star_temp;
177  }
178  if (c_x != NULL) {
179  delete c_x;
180  c_x = NULL;
181  }
182  return ForBESUtils::STATUS_UNDEFINED_FUNCTION;
183 }
