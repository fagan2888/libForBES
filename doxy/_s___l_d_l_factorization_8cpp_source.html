23 Matrix S_LDLFactorization::multiply_AAtr_betaI(Matrix& A, double beta) {
24  size_t n = A.getNrows();
25  size_t m = A.getNcols();
26  Matrix result(n, n, Matrix::MATRIX_SYMMETRIC);
27 
28  for (size_t i = 0; i < n; i++) {
29  for (size_t j = 0; j <= i; j++) {
30  result.set(i, j, 0.0);
31  for (size_t k = 0; k < m; k++) {
32  result.set(i, j, result.get(i, j) + A.get(i, k) * A.get(j, k));
33  }
34  if (i == j) {
35  result.set(i, j, result.get(i, j) + beta);
36  }
37  }
38  }
39  return result;
40 }

47 int S_LDLFactorization::factorize() {
48  if (m_matrix_type == Matrix::MATRIX_SPARSE) {
49  double beta_temp[2];
50  beta_temp[0] = m_beta;
51  beta_temp[1] = 0.0;
52 
53  if (m_matrix->m_sparse == NULL) {
54  m_matrix->_createSparse();
55  }
56 
57  m_matrix->m_sparse->stype = 0;
58  m_factor = cholmod_analyze(m_matrix->m_sparse, Matrix::cholmod_handle());
59  cholmod_factorize_p(m_matrix->m_sparse, beta_temp, NULL, 0, m_factor, Matrix::cholmod_handle());
60  return (m_factor->minor == m_matrix->m_nrows) ? ForBESUtils::STATUS_OK : ForBESUtils::STATUS_NUMERICAL_PROBLEMS;
61  } else if (m_matrix_type == Matrix::MATRIX_DENSE) {
62  /*
63  * We here need to factorize a dense matrix
64  * We distinguish between two cases
65  * 1. A is short (more columns than rows)
66  * 2. A is tall (more rows than columns)
67  */
68  if (m_matrix->getNrows() <= m_matrix->getNcols()) {
69  /* this is a ###SHORT### matrix */
70  /*
71  * Note: here we're creating matrix F on the fly and we're passing
72  * its REFERENCE to LDLFactorization. Eventually, m_delegated_solver will
73  * lose track of the original matrix.
74  */
75  Matrix F = multiply_AAtr_betaI(*m_matrix, m_beta);
76  m_delegated_solver = new LDLFactorization(F);
77  int status = m_delegated_solver->factorize();
78  return status;
79  } else {
80  /* this is a ~~~TALL~~~ matrix */
81  m_matrix->transpose();
82  /*
83  * Note: here we're creating matrix F_tilde on the fly and we're passing
84  * its REFERENCE to LDLFactorization. Eventually, m_delegated_solver will
85  * lose track of the original matrix.
86  */
87  Matrix F_tilde = multiply_AAtr_betaI(*m_matrix, m_beta);
88  m_matrix->transpose();
89  m_delegated_solver = new LDLFactorization(F_tilde);
90  int status = m_delegated_solver->factorize();
91  return status;
92  }
93  } else {
94  throw std::invalid_argument("[uoe] Unsupported operation");
95  }
96 }

98 int S_LDLFactorization::solve(Matrix& rhs, Matrix& solution) {
99  solution = Matrix(rhs.m_nrows, rhs.m_ncols);
100  if (m_matrix_type == Matrix::MATRIX_SPARSE) {
101  if (m_factor == NULL) {
102  throw std::invalid_argument(__FCT_MISS_EXCPT);
103  }
104  cholmod_dense *b;
105  cholmod_dense *x;
106  b = cholmod_allocate_dense(rhs.m_nrows, rhs.m_ncols, rhs.m_nrows, CHOLMOD_REAL, Matrix::cholmod_handle());
107  b->x = rhs.m_data;
108  x = cholmod_solve(CHOLMOD_A, m_factor, b, Matrix::cholmod_handle());
109  solution.m_delete_data = false;
110  memcpy(solution.m_data, static_cast<double*> (x->x), rhs.m_nrows * rhs.m_ncols * sizeof (double));
111  cholmod_free_dense(&x, Matrix::cholmod_handle());
112  return ForBESUtils::STATUS_OK;
113  } else if (m_matrix_type == Matrix::MATRIX_DENSE) {
114  if (m_delegated_solver == NULL) {
115  throw std::invalid_argument(__FCT_MISS_EXCPT);
116  }
117  if (m_matrix_nrows <= m_matrix_ncols) {
118  /* m_matrix is ###SHORT### and dense */
119  int status = m_delegated_solver->solve(rhs, solution);
120  return status;
121  } else {
122  /* m_matrix is ~~~TALL~~~ and dense */
123  m_matrix->transpose();
124  Matrix temp = (*m_matrix) * const_cast<Matrix&>(rhs);
125  m_matrix->transpose();
126  Matrix c;
127  int status = m_delegated_solver->solve(temp, c);
128  if (status != ForBESUtils::STATUS_OK){
129  return status;
130  }
131  solution = *m_matrix * c - rhs;
132  double beta_inv = -1.0/m_beta;
133  solution *= beta_inv;
134  return ForBESUtils::STATUS_OK;
135  }
136 
137  } else {
138  throw std::invalid_argument("[uoe] Unsupported operation");
139  }
140 }
