48 void checkConstructorArguments(const Matrix& Q, const Matrix& q, const Matrix& A, const Matrix& b) {
49  if (Q.getNrows() != Q.getNcols()) {
50  throw std::invalid_argument("Matrix Q is not square");
51  }
52  if (!q.isColumnVector()) {
53  throw std::invalid_argument("q is not a column vector");
54  }
55  if (!b.isColumnVector()) {
56  throw std::invalid_argument("b is not a column vector");
57  }
58  if (Q.getNcols() != q.getNrows()) {
59  throw std::invalid_argument("Q and q have incompatible dimensions");
60  }
61  if (Q.getNcols() != A.getNcols()) {
62  throw std::invalid_argument("Q and A have incompatible dimensions");
63  }
64  if (A.getNrows() != b.getNrows()) { // A and b must have the same number of rows
65  throw std::invalid_argument("A and b have incompatible dimensions");
66  }
67 }

69 QuadOverAffine::QuadOverAffine(Matrix& Q, Matrix& q, Matrix& A, Matrix& b) {
70  checkConstructorArguments(Q, q, A, b);
71 
72  m_F = NULL;
73  m_Fsolver = NULL;
74  m_sigma = NULL;
75 
76  this->m_Q = &Q;
77  this->m_q = &q;
78  this->m_A = &A;
79  this->m_b = &b;
80  size_t n = Q.getNrows();
81  size_t s = A.getNrows();
82  size_t nF = n + s;
83  if (Q.getType() == Matrix::MATRIX_DENSE) {
84  m_F = new Matrix(nF, nF, Matrix::MATRIX_DENSE);
85  /*
86  * F = [Q * ; * *]
87  */
88  for (size_t i = 0; i < n; i++) {
89  for (size_t j = 0; j < n; j++) {
90  m_F->set(i, j, Q.get(i, j));
91  }
92  }
93  /*
94  * F = [Q A' ; A *]
95  */
96  for (size_t i = 0; i < s; i++) {
97  for (size_t j = 0; j < n; j++) {
98  m_F->set(i + n, j, A.get(i, j));
99  m_F->set(j, i + n, A.get(i, j));
100  }
101  }
102  }
103  if (m_F != NULL) {
104  m_Fsolver = new LDLFactorization(*m_F);
105  int status = m_Fsolver -> factorize();
106  if (ForBESUtils::STATUS_OK != status) {
107  throw std::invalid_argument("LDL factorization failed for matrix F = [Q A'; A 0] (dense) - invalid arguments Q and A");
108  }
109  }
110  m_sigma = new Matrix(nF, 1, Matrix::MATRIX_DENSE);
111  for (size_t i = 0; i < s; i++) {
112  m_sigma->set(i + n, 0, b.get(i, 0));
113  }
114 }

121 int QuadOverAffine::callConj(Matrix& y, double& f_star, Matrix& grad) {
122  /* Update sigma(x) */
123  for (size_t i = 0; i < m_Q->getNrows(); i++) {
124  m_sigma->getData()[i] = y[i] - m_q->get(i);
125  }
126  /* Solve F*grad = sigma */
127  int status = m_Fsolver->solve(*m_sigma, grad);
128  /* Take the first n elements of grad */
129  grad.reshape(m_Q->getNrows(), 1);
130  /* f_star = grad' * Q * grad / 2.0 */
131  f_star = m_Q->quad(grad);
132  for (size_t i = 0; i < grad.getNrows(); i++) { /* Dot product */
133  f_star += grad[i] * (m_q->get(i) - y[i]);
134  }
135  f_star = -f_star;
136  return status;
137 }
