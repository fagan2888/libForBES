23 LDLFactorization::LDLFactorization(Matrix& matr) : FactoredSolver(matr) {
24  this->LDL = NULL;
25  this->ipiv = NULL;
26  this->m_sparse_ldl_factor = NULL;
27  this->m_matrix_type = m_matrix->getType();
28  this->m_matrix_nrows = m_matrix->getNrows();
29  if (matr.isEmpty()){
30  throw std::invalid_argument("LDL factorization cannot be applied to empty matrices");
31  }
32  if (m_matrix_type == Matrix::MATRIX_LOWERTR) {
33  throw std::invalid_argument("LDL factorization cannot be applied to non-symmetric matrices (e.g., Lower/Upper triangular)");
34  }
35  if (m_matrix_type != Matrix::MATRIX_DENSE &&
36  matr.getType() != Matrix::MATRIX_SYMMETRIC &&
37  matr.getType() != Matrix::MATRIX_SPARSE) { // Only DENSE and SYMMETRIC and SPARSE are supported!
38  throw std::logic_error("This matrix type is not supported by LDLFactorization");
39  }
40  if (matr.getNrows() != matr.getNcols()) {
41  throw std::invalid_argument("Matrix not square");
42  }
43  if (m_matrix_type == Matrix::MATRIX_SPARSE) {
44  m_sparse_ldl_factor = new sparse_ldl_factor;
45  return;
46  }
47  this->LDL = new double[matr.length()];
48  this->ipiv = new int[matr.getNrows()];
49  memcpy(this->LDL, matr.getData(), matr.length() * sizeof (double));
50 }

61 int LDLFactorization::factorize() {
62  int status = ForBESUtils::STATUS_UNDEFINED_FUNCTION;
63  if (this->m_matrix_type == Matrix::MATRIX_DENSE) {
64  status = LAPACKE_dsytrf(LAPACK_COL_MAJOR, 'L', m_matrix_nrows, LDL, m_matrix_nrows, ipiv);
65  } else if (this->m_matrix_type == Matrix::MATRIX_SYMMETRIC) {
66  status = LAPACKE_dsptrf(LAPACK_COL_MAJOR, 'L', m_matrix_nrows, LDL, ipiv);
67  } else if (this->m_matrix_type == Matrix::MATRIX_SPARSE) {
68  // Factorize sparse matrix
69  m_matrix->_createSparse();
70  int * Parent = new int[m_matrix_nrows];
71  int * Lnz = new int[m_matrix_nrows];
72  int * Flag = new int[m_matrix_nrows];
73  m_sparse_ldl_factor->Lp = new int[m_matrix_nrows + 1];
74  ldl_symbolic(m_matrix_nrows,
75  static_cast<int*>(m_matrix->m_sparse->p),
76  static_cast<int*>(m_matrix->m_sparse->i),
77  m_sparse_ldl_factor->Lp,
78  Parent,
79  Lnz,
80  Flag,
81  NULL, NULL);
82  int lnz = m_sparse_ldl_factor->Lp[m_matrix_nrows];
83  int d;
84  m_sparse_ldl_factor->Li = new int[lnz];
85  m_sparse_ldl_factor->Lx = new double[lnz];
86  m_sparse_ldl_factor->D = new double[m_matrix_nrows];
87 
88  double *Y = new double[m_matrix_nrows];
89  int *Pattern = new int[m_matrix_nrows];
90 
91  d = ldl_numeric(m_matrix_nrows,
92  static_cast<int*>(m_matrix->m_sparse->p),
93  static_cast<int*>(m_matrix->m_sparse->i),
94  static_cast<double*>(m_matrix->m_sparse->x),
95  m_sparse_ldl_factor->Lp,
96  Parent,
97  Lnz,
98  m_sparse_ldl_factor->Li,
99  m_sparse_ldl_factor->Lx,
100  m_sparse_ldl_factor->D,
101  Y,
102  Pattern,
103  Flag,
104  NULL, NULL);
105 
106  delete[] Parent;
107  delete[] Pattern;
108  delete[] Y;
109  delete[] Flag;
110  delete[] Lnz;
111 
112  return d == m_matrix_nrows ? ForBESUtils::STATUS_OK : ForBESUtils::STATUS_NUMERICAL_PROBLEMS;
113 
114  } else {
115  throw std::invalid_argument("This matrix type is not supported by LDLFactorization");
116  }
117  return status;
118 }

120 int LDLFactorization::solve(Matrix& rhs, Matrix& solution) {
121  solution = Matrix(m_matrix_nrows, 1, Matrix::MATRIX_DENSE); // solution = rhs (DENSE)
122  for (size_t i = 0; i < m_matrix_nrows; i++) {
123  solution.set(i, 0, rhs.get(i, 0));
124  }
125  int status = ForBESUtils::STATUS_OK;
126  if (Matrix::MATRIX_DENSE == this->m_matrix_type) {
127  status = LAPACKE_dsytrs(LAPACK_COL_MAJOR, 'L', m_matrix_nrows, 1, LDL, m_matrix_nrows, ipiv, solution.getData(), m_matrix_nrows);
128  } else if (Matrix::MATRIX_SYMMETRIC == this->m_matrix_type) {
129  status = LAPACKE_dsptrs(LAPACK_COL_MAJOR, 'L', m_matrix_nrows, 1, LDL, ipiv, solution.getData(), m_matrix_nrows);
130  } else if (Matrix::MATRIX_SPARSE == this->m_matrix_type) {
131  double * b = solution.getData();
132  ldl_lsolve(m_matrix_nrows, b,
133  m_sparse_ldl_factor->Lp,
134  m_sparse_ldl_factor->Li,
135  m_sparse_ldl_factor->Lx);
136  ldl_dsolve(m_matrix_nrows, b, m_sparse_ldl_factor->D);
137  ldl_ltsolve(m_matrix_nrows, b, m_sparse_ldl_factor->Lp, m_sparse_ldl_factor->Li, m_sparse_ldl_factor->Lx);
138  status = ForBESUtils::STATUS_OK;
139  }
140  return status;
141 }
