28  // store current point temporarily
29  Matrix * temp = new Matrix(*m_cache.get_point());
30  // extrapolate if not first iterate
31  if (m_previous != NULL) {
32  // y = x + k/(k+2) (x - x')
33  // = (2k+2)/(k+2) x - k/(k+2) x'
34  Matrix::add(*m_cache.get_point(), -(1.0 * m_it) / (m_it + 2), *m_previous, (2.0 * m_it + 2.0) / (m_it + 2));
35  // tell FBCache that the point has changed
36  m_cache.reset();
37  // delete previously allocated 'm_previous'

39  }
