38 int OpReverseVector::call(Matrix& y, double alpha, Matrix& x, double gamma) {
39  if (y.getNrows() == 0) {
40  y = Matrix(x.getNrows(), 1);
41  }
42  bool is_gamma_zero = (std::abs(gamma) < std::numeric_limits<double>::epsilon());
43  size_t n = x.length();
44  for (size_t i = 0; i < n / 2; ++i) {
45  double temp;
46  temp = x.get(n - i - 1, 0);
47  if (is_gamma_zero) {
48  y[n - i - 1] = alpha * x[i];
49  y[i] = alpha * temp;
50  } else {
51  y[n - i - 1] = gamma * y[n - i - 1] + alpha * x[i];
52  y[i] = gamma * y[i] + alpha * temp;
53  }
54  }
55  if (n % 2 == 1) {
56  size_t middle_idx = n / 2;
57  if (is_gamma_zero) {
58  y[middle_idx] = alpha * x[middle_idx];
59  } else {
60  y[middle_idx] = gamma * y[middle_idx] + alpha * x[middle_idx];
61  }
62  }
63  return ForBESUtils::STATUS_OK;
64 }
