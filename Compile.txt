4. Netbeans compilation commands:

SOURCE_DIR="/home/chung/Documents/MATLAB/ForBES/source/"
OBJ_DIR="build/Debug/GNU-Linux-x86/"

# Compiler options
COPS="-fprofile-arcs -ftest-coverage   -c -g -DUSE_LIBS"

FORBES_OPTIONS="-fprofile-arcs -ftest-coverage   -c -g -DUSE_LIBS -I/home/chung/Documents/MATLAB/SuiteSparse/CHOLMOD/Include  \
-I/home/chung/Documents/MATLAB/SuiteSparse/SuiteSparse_config  \
-I/home/chung/Documents/MATLAB/SuiteSparse/LDL/Include -MMD -MP -MF"

# Compile ForBESUtils.cpp
g++  ${FORBES_OPTIONS} "${OBJ_DIR}ForBESUtils.o.d" -o ${OBJ_DIR}ForBESUtils.o ${SOURCE_DIR}ForBESUtils.cpp
